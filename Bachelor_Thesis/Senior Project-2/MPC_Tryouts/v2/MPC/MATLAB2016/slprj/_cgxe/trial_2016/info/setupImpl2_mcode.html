148        function setupImpl(obj)
149            % Implement tasks that need to be performed only once, 
150            % such as pre-computed constants.
151            
152            % CasADi v3.5.1
153            addpath('C:\Program Files\casadi-windows-matlabR2016a-v3.5.1')
154            import casadi.*
155            
156            % Updated Variables at Each Time-Step
157            obj.u0 = zeros(obj.N, 1);
158            obj.X0 = zeros(obj.N + 1, 1);
159
160            % Constants for Vehicle Dynamics
161            obj.Rtire  =  0.354;  obj.G      =  9.81;   
162            obj.mu     =  0.09;   obj.rho    =  1.225; 
163            obj.Cd     =  0.55;   obj.A      =  5.1;    
164            obj.mss    =  2100;   obj.ng     =  0.85;   
165            obj.Im     =  0.18;   obj.Gb     =  18.5;   
166
167            % Symbols for States, Controls and Changing Constraints
168            obj.states = SX.sym('v'); n_states = length(obj.states);
169            obj.controls = SX.sym('trq'); n_controls = length(obj.controls);
170
171            % Variables for Vehicle Dynamics
172            obj.alpha     =  atan(0 / 100);
173            obj.Faero     =  0.5*obj.rho*obj.A*obj.Cd*(obj.states^2);                     
174            obj.Frub      =  obj.mu*obj.mss*obj.G;                                   
175            obj.Ftire     =  obj.controls*obj.Gb / obj.Rtire;
176            obj.dvdt      =  (obj.Ftire - obj.Frub - obj.Faero - obj.mss*obj.G*sin(obj.alpha)...
177                - obj.Fbrake)/(obj.mss + obj.Im*obj.Gb^2 / (obj.ng*obj.Rtire^2));
178
179            % Objective Function and Constraints Vector
180            obj.J = 0; obj.g = [];                                              
181
182            % Non-Linear Mapping Function f(x,u) and System R.H.S
183            obj.f = Function('f', {obj.states, obj.controls}, {obj.dvdt});
184            obj.U = SX.sym('U', n_controls, obj.N);       
185            obj.X = SX.sym('X', n_states, (obj.N + 1));    
186            obj.P = SX.sym('P', n_states + obj.N*(n_states + n_controls));  
187            
188            % Computation of the Objective Function
189                        
190            % Caching of the System Object Data in Local Variable Before 
191            % Multiple Access in Each Iteration in a Loop
192                        
193            for k = 1:obj.N
194                current_state = obj.X(k); current_control = obj.U(k);
195                
196                %if k == obj.N
197                %    next_control = U(k);
198                %else
199                %    next_control = U(k+1);
200                %end
201                
202                obj.J = obj.J + ((current_state - obj.P(2*k))'*obj.Q*(current_state - obj.P(2*k))) + ...
203                          ((current_control - obj.P(2*k+1))'*obj.R*(current_control - obj.P(2*k+1))); %+ ...
204                          %((next_control-current_control)*obj.S*(next_control-current_control));
205            end
206
207            % Non-Linear Programming Problem Initialization
208            obj.opt_variables = [reshape(obj.X, obj.N+1, 1);reshape(obj.U, obj.N, 1)];
209            %obj.nlp_prob = struct('f', obj.J, 'x', obj.opt_variables, 'g', obj.g, 'p', obj.P);
210
211            obj.opts = struct;
212            obj.opts.ipopt.max_iter = 2000;
213            obj.opts.ipopt.print_level =0;%0,3
214            obj.opts.print_time = 0;
215            obj.opts.ipopt.acceptable_tol =1e-8;
216            obj.opts.ipopt.acceptable_obj_change_tol = 1e-6;
217
218            rpm = (obj.states/obj.Rtire)*(60/(2*pi))*obj.Gb; % revolutions per minute(rpm)
219
220            obj.rpm_calc  = Function('rpm_calc', {obj.states},{rpm});
221            %obj.solver    = nlpsol('solver', 'ipopt', obj.nlp_prob, obj.opts);
222        end

224        function [trq_req] = stepImpl(obj, cur_sim_tim, cur_speed,...
225                ref_speed, ref_control, grade)      
226            import casadi.* 
227            inst = uint16((cur_sim_tim/obj.T) + 1); % kth instance of simulation
228            
229            if inst > length(ref_speed) - obj.N
230                xs = ref_speed(inst:end);
231                cs = ref_control(inst:end);
232                gs = grade(inst:end);
233                
234                check = (length(ref_speed) - inst + 1); %to check how many 
235                %reference velocity points are supposed to be added to have
236                %exact N points at every step
237                
238                for iter = check:obj.N-1 %array completion
239                    xs(length(xs)+1) = xs(end); 
240                    cs(length(cs)+1) = cs(end);
241                    gs(length(gs)+1) = gs(end);
242                end
243            else
244                xs = ref_speed(inst:(inst+obj.N-1));   % reference speed
245                cs = ref_control(inst:(inst+obj.N-1)); % reference control
246                gs = grade(inst:(inst+obj.N-1));       % grade data
247            end
248                        
249%             motor_inv_max_trq_output   = [250,250,249,243,215,162,132.5,112.5,97,84,79.6];
250%             motor_inv_max_rpm_input    = [0,4000,5000,5500,6000,7000,8000,9000,10000,11000,12000];         
251%             
252%             maxAvailableTrq = interp1(motor_inv_max_rpm_input,motor_inv_max_trq_output,full(obj.rpm_calc(cur_speed)));
253%             
254%             if  isnan(maxAvailableTrq)
255%                 maxAvailableTrq = 250; 
256%             end
257            
258            maxAvailableTrq = 250;
259            
260            v_max    = 55;  v_min   = 0; 
261            trq_max =  maxAvailableTrq; trq_min = -maxAvailableTrq;
262            
263            calc_constraints_vect(obj,gs);
264            obj.nlp_prob = struct('f', obj.J, 'x', obj.opt_variables, 'g', obj.g, 'p', obj.P);
265            obj.solver   = nlpsol('solver', 'ipopt', obj.nlp_prob, obj.opts);
266            
267            args = struct;
268            
269            %Equality Constraints
270            args.lbg(1:obj.N+1) = 0;    
271            args.ubg(1:obj.N+1) = 0;     
272
273            %States Upper and Lower Bounds
274            args.lbx(1:obj.N+1) = v_min; %velocity lower bound
275            args.ubx(1:obj.N+1) = v_max; %velocity upper bound
276
277            %Controls Upper and Lower Bounds
278            args.lbx((obj.N+1)+1:(2*obj.N)+1) = trq_min; %torque lower bound
279            args.ubx((obj.N+1)+1:(2*obj.N)+1) = trq_max; %torque upper bound
280            
281            args.p(1) = cur_speed; % initial condition of the robot posture
282            for k = 1:obj.N %new - set the reference to track
283                args.p(2*k)   = xs(k);
284                args.p(2*k+1) = cs(k);
285            end
286            
287            args.x0  = [reshape(obj.X0',(obj.N+1),1);reshape(obj.u0',obj.N,1)];
288            
289            sol = obj.solver('x0', args.x0, 'lbx', args.lbx, 'ubx', args.ubx,...
290            'lbg', args.lbg, 'ubg', args.ubg,'p',args.p); 
291            
292            u = reshape(full(sol.x((obj.N+1)+1:end))',1,obj.N)'; %control actions for the given prediction horizon
293            obj.X0 = full(sol.x(1:(obj.N+1))); %predicted states 
294            trq_req = u(1); %first control action is output to be applied
295            obj.u0 = [u(2:size(u,1));u(size(u,1))]; %shift controls to initialize the next step
296            obj.X0 = [obj.X0(2:end);obj.X0(end)]; % shift trajectory to initialize the next step
297        end

299        function calc_constraints_vect(obj, grade_data)
300            import casadi.*            
301            g_rep = [];             Ftire_rep = obj.Ftire;  Frub_rep  = obj.Frub;   Fbrake_rep = obj.Fbrake;
302            ng_rep = obj.ng;        Faero_rep = obj.Faero;  mss_rep = obj.mss;      U_rep = obj.U;   
303            G_rep = obj.G;          Im_rep = obj.Im;        Gb_rep = obj.Gb;        N_rep = obj.N;
304            Rtire_rep = obj.Rtire;  X_rep = obj.X;          T_rep = obj.T;          P_rep = obj.P;
305            
306            init_state = X_rep(1);                    % initial state
307            g_rep = [g_rep; (init_state - P_rep(1))]; % initial condition constraints
308            for k = 1:N_rep
309                alpha_rep = atan(grade_data(k) / 100);
310                dvdt_rep = (Ftire_rep - Frub_rep - Faero_rep - mss_rep*G_rep*sin(alpha_rep) - Fbrake_rep)/(mss_rep + Im_rep*Gb_rep^2 / (ng_rep*Rtire_rep^2));
311                func = Function('func', {obj.states, obj.controls}, {dvdt_rep});
312                
313                current_state = X_rep(k); current_control = U_rep(k);
314                f_value  = func(current_state,current_control);
315                next_state = X_rep(k+1);  
316                next_state_pred = current_state + (T_rep*f_value);
317                g_rep = [g_rep; next_state - next_state_pred];
318            end
319            obj.alpha = alpha_rep;
320            obj.dvdt  = dvdt_rep;
321            obj.g = g_rep;
322        end
