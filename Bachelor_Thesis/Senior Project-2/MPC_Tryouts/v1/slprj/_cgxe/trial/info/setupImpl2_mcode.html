 83        function setupImpl(obj)
 84            % Implement tasks that need to be performed only once, 
 85            % such as pre-computed constants.
 86            
 87            % CasADi v3.5.1
 88        addpath('C:\Program Files\casadi-windows-matlabR2016a-v3.5.1')
 89        import casadi.*
 90
 91        % Constants for MPC Design
 92        T = 0.2; % sampling time(s)
 93        N = 5;  % prediction horizon
 94
 95        % Weighing Matrices for States and Controls
 96        Q = 5; % multiplied by (current state - reference state)^2
 97        R = 0.0001; % multiplied by (current control - reference control)^2
 98        S = 0.01; % multiplied by (next control - current control)^2
 99
100        % Conversion Metrics
101        % 1 rad/s = 1/2*pi Hz = 60/2*pi RPM
102        % 1 m/s   = 3.6 km/h
103        % 1 hP    = 745.7 W
104
105        % Constants for Vehicle Dynamics
106        Rtire  =  0.354;  % radius of tire(m)
107        G      =  9.81;   % earth's gravitational constant(m/s2)
108        mu     =  0.09;   % road friction(N) 
109        rho    =  1.225;  % density of air as a fluid(kg/m3)
110        Cd     =  0.55;   % air drag coefficient
111        A      =  5.1;    % vehicle frontal area(m2)
112        mss    =  2100;   % vehicle mass(kg)
113        ng     =  0.85;   % tire-specific constant
114        Im     =  0.18;   % motor inertia(kg*m2)
115        Gb     =  18.5;   % total gear ratio
116
117        % State Variables and Vectors
118        v = SX.sym('v');      % velocity(m/s)
119        states = v;       
120        n_states = length(states);
121
122        % Control Variables, Vectors
123        trq = SX.sym('trq');  % torque(N*m)
124        controls = trq; 
125        n_controls = length(controls);
126
127        % Variables for Vehicle Dynamics                                           
128        Fbrake    =  0;                             % braking force(N)
129        Grade     =  0;                             % grade(%)
130        alpha     =  atan(Grade/100);               % degree of slope(rad)
131        Faero     =  0.5*rho*A*Cd*(states^2);       % drag force against vehicle(N)
132        Frub      =  mu*mss*G;                      % rubbing force against tires(N)
133        Ftire     =  controls*Gb/Rtire;             % driving force(N)
134        dvdt      =  (Ftire - Frub - Faero - mss*G*sin(alpha) - Fbrake)/...
135            (mss + Im*Gb^2/(ng*Rtire^2)); %acceleration(m/s2)
136
137        J = 0; g = [];
138
139        % Non-Linear Mapping Function f(x,u) and System R.H.S
140        rhs_initial = (Ftire - Frub - mss*G*sin(alpha) - Fbrake)/(mss + Im*Gb^2/(ng*Rtire^2));
141        rhs_continued = dvdt;
142        f1 = Function('f1',{states,controls},{rhs_initial});
143        f2 = Function('f2',{states,controls},{rhs_continued});
144
145        U = SX.sym('U',n_controls,N);        % decision variables (controls)
146        X = SX.sym('X',n_states,(N+1));      % a vector that represents the states over the optimization problem
147        P = SX.sym('P',n_states + N*(n_states+n_controls)); % parameters (which include the initial state and 
148        %the reference along the predicted trajectory (reference states and reference controls))
149
150        % Computation of the Objective Function and Constraints Vector
151
152        init_state = X(1);       % initial state
153        g = [g;(init_state-P(1))]; % initial condition constraints
154        for k=1:N
155            current_state = X(k); current_control = U(k);
156            if k == N
157                next_control = U(k);
158            else
159                next_control = U(k+1);
160            end
161            J = J+(current_state-P(2*k))*Q*(current_state-P(2*k)) + ...
162                      (current_control-P(2*k+1))'*R*(current_control-P(2*k+1)) + ...
163                      ((next_control-current_control)*S*(next_control-current_control));
164            if k == 1
165                f_value  = f1(current_state,current_control);
166            else
167                f_value  = f2(current_state,current_control);
168            end
169            next_state = X(k+1);  
170            next_state_pred = current_state + (T*f_value);
171            g = [g;next_state - next_state_pred];
172        end
173
174        % Non-Linear Programming Problem Initialization
175        opt_variables = [reshape(X,N+1,1);reshape(U,N,1)];
176        nlp_prob = struct('f', J, 'x', opt_variables, 'g', g, 'p', P);
177
178        opts = struct;
179        opts.ipopt.max_iter = 2000;
180        opts.ipopt.print_level =0;%0,3
181        opts.print_time = 0;
182        opts.ipopt.acceptable_tol =1e-8;
183        opts.ipopt.acceptable_obj_change_tol = 1e-6;
184
185        rpm           =  (states/Rtire)*(60/(2*pi))*Gb; % revolutions per minute(rpm)
186        obj.rpm_calc  = Function('rpm_calc', {states},{rpm});
187        
188        obj.solver = nlpsol('solver', 'ipopt', nlp_prob, opts);
189        
190        obj.sample_time = T;
191        obj.pred_horizon = N;
192        end
193

194        function [trq_req,X0_out,u0_out] = stepImpl(obj, cur_sim_tim, cur_speed,... 
195                ref_speed, ref_control, prev_X0, prev_u0)
196            
197            T = obj.sample_time;  %sample time
198            N = obj.pred_horizon; %prediction horizon       
199            
200            inst = (cur_sim_tim/T) + 1; % kth instance of simulation
201            x0 = cur_speed;
202            
203            if inst == 1 
204                u0 = zeros(N,1);
205                X0 = repmat(x0,1,N+1)';
206            else
207                u0 = prev_u0;        
208                X0 = prev_X0;
209            end
210
211            if inst > length(ref_speed) - N
212                xs = ref_speed(inst:end);
213                cs = ref_control(inst:end);
214                
215                check = (length(ref_speed) - inst + 1); %to check how many 
216                %reference velocity points are supposed to be added to have
217                %exact N points at every step
218                
219                for iter = check:N-1 %array completion
220                    xs(length(xs)+1) = xs(end); 
221                    cs(length(cs)+1) = cs(end);
222                end
223            else
224                xs = ref_speed(inst:inst+N-1);  % reference speed
225                cs = ref_control(inst:inst+N-1);% reference control
226            end
227            
228            motor_inv_max_trq_output   = [250,250,249,243,215,162,132.5,112.5,97,84,79.6];
229            motor_inv_max_rpm_input    = [0,4000,5000,5500,6000,7000,8000,9000,10000,11000,12000];
230            maxAvailableTrq = interp1(motor_inv_max_rpm_input,motor_inv_max_trq_output,full(obj.rpm_calc(x0)));
231            
232            v_max    = 40;  v_min   = 0; 
233            trq_max =  maxAvailableTrq; trq_min = -maxAvailableTrq;
234            
235            args = struct;
236            
237            %Equality Constraints
238            args.lbg(1:N+1) = 0;    
239            args.ubg(1:N+1) = 0;     
240
241            %States Upper and Lower Bounds
242            args.lbx(1:N+1) = v_min; %velocity lower bound
243            args.ubx(1:N+1) = v_max; %velocity upper bound
244
245            %Controls Upper and Lower Bounds
246            args.lbx((N+1)+1:(2*N)+1) = trq_min; %torque lower bound
247            args.ubx((N+1)+1:(2*N)+1) = trq_max; %torque upper bound
248
249            args.p(1) = x0; % initial condition of the robot posture
250            for k = 1:N %new - set the reference to track
251                args.p(2*k)   = xs(k);
252                args.p(2*k+1) = cs(k);
253            end
254            
255            args.x0  = [reshape(X0',(N+1),1);reshape(u0',N,1)];
256            
257            sol = obj.solver('x0', args.x0, 'lbx', args.lbx, 'ubx', args.ubx,...
258            'lbg', args.lbg, 'ubg', args.ubg,'p',args.p); 
259            
260            u = reshape(full(sol.x((N+1)+1:end))',1,N)'; %control actions for the given prediction horizon
261            X0 = full(sol.x(1:(N+1))); %predicted states 
262
263            trq_req = u(1); %first control action is output to be applied
264            u0 = [u(2:size(u,1));u(size(u,1))]; %shift controls to initialize the next step
265            X0 = [X0(2:end);X0(end)]; % shift trajectory to initialize the next step
266            
267            u0_out = u0;
268            X0_out = X0;
269        end
