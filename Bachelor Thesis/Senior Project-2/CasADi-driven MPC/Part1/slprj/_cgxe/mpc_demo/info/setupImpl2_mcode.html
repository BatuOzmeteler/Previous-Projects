 56        function setupImpl(obj,~,~)
 57            % Implement tasks that need to be performed only once, 
 58            % such as pre-computed constants.
 59            
 60            import casadi.*
 61
 62            T = 10; % Time horizon
 63            N = 20; % number of control intervals
 64
 65            % Declare model variables
 66            x1 = SX.sym('x1');
 67            x2 = SX.sym('x2');
 68            x = [x1; x2];
 69            u = SX.sym('u');
 70
 71            % Model equations
 72            xdot = [(1-x2^2)*x1 - x2 + u; x1];
 73
 74            % Objective term
 75            L = x1^2 + x2^2 + u^2;
 76
 77            % Continuous time dynamics
 78            f = casadi.Function('f', {x, u}, {xdot, L});
 79
 80            % Formulate discrete time dynamics
 81            % Fixed step Runge-Kutta 4 integrator
 82            M = 4; % RK4 steps per interval
 83            DT = T/N/M;
 84            f = Function('f', {x, u}, {xdot, L});
 85            X0 = MX.sym('X0', 2);
 86            U = MX.sym('U');
 87            X = X0;
 88            Q = 0;
 89            for j=1:M
 90               [k1, k1_q] = f(X, U);
 91               [k2, k2_q] = f(X + DT/2 * k1, U);
 92               [k3, k3_q] = f(X + DT/2 * k2, U);
 93               [k4, k4_q] = f(X + DT * k3, U);
 94               X=X+DT/6*(k1 +2*k2 +2*k3 +k4);
 95               Q = Q + DT/6*(k1_q + 2*k2_q + 2*k3_q + k4_q);
 96            end
 97            F = Function('F', {X0, U}, {X, Q}, {'x0','p'}, {'xf', 'qf'});
 98
 99            % Start with an empty NLP
100            w={};
101            w0 = [];
102            lbw = [];
103            ubw = [];
104            J = 0;
105            g={};
106            lbg = [];
107            ubg = [];
108
109            % "Lift" initial conditions
110            X0 = MX.sym('X0', 2);
111            w = {w{:}, X0};
112            lbw = [lbw; 0; 1];
113            ubw = [ubw; 0; 1];
114            w0 = [w0; 0; 1];
115
116            % Formulate the NLP
117            Xk = X0;
118            for k=0:N-1
119                % New NLP variable for the control
120                Uk = MX.sym(['U_' num2str(k)]);
121                w = {w{:}, Uk};
122                lbw = [lbw; -1];
123                ubw = [ubw;  1];
124                w0 = [w0;  0];
125
126                % Integrate till the end of the interval
127                Fk = F('x0', Xk, 'p', Uk);
128                Xk_end = Fk.xf;
129                J=J+Fk.qf;
130
131                % New NLP variable for state at end of interval
132                Xk = MX.sym(['X_' num2str(k+1)], 2);
133                w = {w{:}, Xk};
134                lbw = [lbw; -0.25; -inf];
135                ubw = [ubw;  inf;  inf];
136                w0 = [w0; 0; 0];
137
138                % Add equality constraint
139                g = {g{:}, Xk_end-Xk};
140                lbg = [lbg; 0; 0];
141                ubg = [ubg; 0; 0];
142            end
143
144            % Create an NLP solver
145            prob = struct('f', J, 'x', vertcat(w{:}), 'g', vertcat(g{:}));
146            options = struct('ipopt',struct('print_level',0),'print_time',false);
147            solver = nlpsol('solver', 'ipopt', prob, options);
148
149            obj.casadi_solver = solver;
150            obj.x0 = w0;
151            obj.lbx = lbw;
152            obj.ubx = ubw;
153            obj.lbg = lbg;
154            obj.ubg = ubg;
155        end

157        function u = stepImpl(obj,x,t)
158            disp(t)
159            tic
160            w0 = obj.x0;
161            lbw = obj.lbx;
162            ubw = obj.ubx;
163            solver = obj.casadi_solver;
164            lbw(1:2) = x;
165            ubw(1:2) = x;
166            sol = solver('x0', w0, 'lbx', lbw, 'ubx', ubw,...
167                        'lbg', obj.lbg, 'ubg', obj.ubg);
168  
169            u = full(sol.x(3));
170            toc
171        end
